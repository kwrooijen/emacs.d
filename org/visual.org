** Config :config:
#+BEGIN_SRC emacs-lisp
(defun kwrooijen/helm-dim-all-initialize ()
  "Dim all buffers except during `helm-occur'."
  (dimmer-dim-all)
  (when (string= helm-buffer "*helm occur*")
    (dimmer-restore-buffer (current-buffer))))

;;;###autoload
(define-minor-mode kwrooijen/helm-dim-mode
  "Toggle dimming adjustments for Helm."
  :global t
  (if kwrooijen/helm-dim-mode
      (progn
        (add-to-list 'dimmer-exclusion-regexp-list "^\\*[hH]elm.*\\*$")
        (add-hook 'helm-after-initialize-hook #'kwrooijen/helm-dim-all-initialize)
        (add-hook 'helm-cleanup-hook #'dimmer-restore-all))
    (remove-hook 'helm-after-initialize-hook #'kwrooijen/helm-dim-all-initialize)
    (remove-hook 'helm-cleanup-hook #'dimmer-restore-all)))

(kwrooijen/helm-dim-mode)
#+END_SRC
